country_event = {
	id = yugreworkevent.1
	title = yugreworkevent.1.t
	desc = yugreworkevent.1.d
	picture = GFX_report_event_election_vote

	fire_only_once = no
	is_triggered_only = yes

	trigger = {
		tag = YUG
		has_completed_focus = YUG_rally_of_youth
		date > 1936.5.25
	}
	
	option = {
		name = yugreworkevent.1.a
		ai_chance = {
			factor = 4000
		}
		add_political_power = 10
		add_stability = 0.01
			if = {
				limit = {
					has_war = yes
				}
				add_war_support = 0.05
					hidden_effect = {
						country_event = {
							name = yugreworkevent.1
							days = 365
						}
					}
			}
			else = {
				add_stability = 0.05
				hidden_effect = {
					country_event = {
						name = yugreworkevent.1
						days = 365
					}
				}
			}
	}
}

	title = yugreworkevent.5.t
	desc = yugreworkevent.5.d
	picture = GFX_report_event_election_vote

	#is_triggered_only = no
	#fire_only_once = not

	trigger = {
		has_idea = YUG_idea_wartime_alignment0
		tag = YUG
		has_war = no
	}

	mean_time_to_happen = {
		days = 2
	}

	option = {
		name = yugreworkevent.5.a
		if = {
			limit = {
				has_idea = YUG_idea_wartime_alignment0
			}
			swap_ideas = {
				remove_idea = YUG_idea_wartime_alignment0
				add_idea = YUG_idea_peacetime_alignment0
			}
		}
	}
}
