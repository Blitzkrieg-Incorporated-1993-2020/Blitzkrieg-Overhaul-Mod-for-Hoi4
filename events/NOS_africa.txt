country_event = {
	id = NOS_africa.1
	title = NOS_africa.1.t	# Freedom for BOT
	desc = NOS_africa.1.desc
	picture = GFX_report_event_african_soldiers
	
	is_triggered_only = yes
	option = { #puppet new state
		name = NOS_africa.1.a
		ai_chance = {
			base = 100
		}
		BOT = { 
			if = {
				limit = { 
					ETH = { 
						controls_state = 542
					}
				}
					transfer_state = 542
			}
		}
		ETH = { puppet = BOT }
	}

	option = { #full freedom
		name = NOS_africa.1.b
		ai_chance = {
			base = 0
		}
		BOT = { 
			if = {
				limit = { 
					ETH = { 
						controls_state = 542
						} 

					}
					transfer_state = 542
				}
			hidden_effect = {
				set_politics = {
					ruling_party = communism
					elections_allowed = no
				}
				add_popularity = {
					ideology = communism
					popularity = 0.75
				}
			}	
		}
		ETH = { add_to_faction = BOT }		
	}

	option = { #refuse them because of reasons
		name = NOS_africa.1.c
		ai_chance = {
			base = 0
		}
		add_political_power = -70
	}	
}
