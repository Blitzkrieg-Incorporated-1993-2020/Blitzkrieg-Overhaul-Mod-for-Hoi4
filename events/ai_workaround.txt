country_event = {		## War Bonds
    id = ai.1
    hidden = yes
    
	mean_time_to_happen = { days = 2 }
	
    trigger = { 
    	any_country = {
    		is_ai = yes
    		has_war = yes
			has_political_power > 150
    		OR = {
    			AND = {
    				NOT = { has_idea = bonds_basic }
    				NOT = { has_idea = bonds_war_public }
    				NOT = { has_idea = bonds_print }
					OR = {
						has_war = yes
						has_government = fascism
						has_government = communism
					}
    			}
    			AND = {
    				NOT = { has_idea = bonds_war_public }
    				NOT = { has_idea = bonds_print }
					has_war = yes
					OR = {
						has_country_flag = is_in_world_war
						surrender_progress > 0.1
						has_country_flag = missing_core_territory
					}
					OR = {
						has_idea = unity_broken_nation
						has_idea = unity_divided_nation
						has_idea = unity_united_nation
					}
    			}
    			AND = {
    				NOT = { has_idea = bonds_print }
					has_war = yes
					OR = {
						surrender_progress > 0.2
						has_country_flag = missing_core_territory
					}
					OR = {
						has_idea = unity_divided_nation
						has_idea = unity_united_nation
					}
    			}
    		}
    	}
    }
    
    option = {
    	every_country = {
    		limit = {
	    		is_ai = yes
	    		has_war = yes
    			has_political_power > 150
				NOT = { has_idea = bonds_basic }
				NOT = { has_idea = bonds_war_public }
				NOT = { has_idea = bonds_print }
				OR = {
					has_war = yes
					has_government = fascism
					has_government = communism
				}
	    	}
	        add_ideas = bonds_basic
			add_political_power = -150
    	}
    	every_country = {
    		limit = {
	    		is_ai = yes
	    		has_war = yes
    			has_political_power > 150
				NOT = { has_idea = bonds_war_public }
				NOT = { has_idea = bonds_print }
				has_war = yes
				OR = {
					has_country_flag = is_in_world_war
					surrender_progress > 0.1
					has_country_flag = missing_core_territory
				}
				OR = {
					has_idea = unity_broken_nation
					has_idea = unity_divided_nation
					has_idea = unity_united_nation
				}
	    	}
	        add_ideas = bonds_war_public
			add_political_power = -150
    	}
    	every_country = {
    		limit = {
	    		is_ai = yes

				NOT = { has_idea = bonds_print }
