# Soviet Ultimatum
country_event = {
	id = finland.1
	title = finland.1.t
	desc = finland.1.d
	picture = GFX_report_event_soviet_tanks_snow
	
	is_triggered_only = yes

	option = { # Accept demands
		name = finland.1.a
		ai_chance = { 
			factor = 25
			modifier = {
				factor = 0
				OR = {
					any_other_country = {
						is_major = yes
						OR = {
							is_in_faction_with = ROOT
							has_guaranteed = ROOT
						}
					}
					has_idea = anti_soviet_pact
					AND = {
						is_historical_focus_on = yes
						date < 1941.1.1
					}
				}
			}
			modifier = {
				factor = 1000
				AND = {
					is_historical_focus_on = yes
					date > 1941.1.1
				}
			}
		}
		SOV = {
			country_event = { days = 5 id = soviet.12 }
			effect_tooltip = {
				transfer_state = 146
			}
		}
	}
	option = { # Refuse
		name = finland.1.b
		ai_chance = { factor = 75 }
		SOV = {
			country_event = { days = 5 id = soviet.13 }
			effect_tooltip = {
				create_wargoal = {
					type = take_state
					target = ROOT
					generator = { 146 }
				}
			}
		}
	}
}

# Swedish Support in Winter War
country_event = {
	id = finland.3
	title = finland.3.t
	desc = finland.3.d
	picture = GFX_report_event_finnish_letter
	
	is_triggered_only = yes

	option = {
		name = finland.3.a

		ai_chance = {
			factor = 10 
			modifier = {
				factor = 10
				OR = {
					has_war = yes
					OR = { has_government = unionism has_government = communism }
				}
			}
		}

		FIN = {
			country_event = { id = finland.4 days = 3 }
			add_opinion_modifier = {
				target = ROOT
				modifier = refused_help
			}
		}
	}

	option = {
		name = finland.3.b

		ai_chance = {
			factor = 25
			modifier = {
				factor = 5							
				has_democratic_government = yes
				FIN = {
				 has_democratic_government = yes
				}
			}
		}

		trigger = {
			AND = {
				has_war = no
				NOT = {
					OR = { has_government = unionism has_government = communism }
				}
			}	
			has_manpower > 8402
		}
		add_manpower = -8402
		if = {
			limit = {
				has_equipment = {
					infantry_equipment > 500
				}
			}
			send_equipment = {
				target = FIN
				type = infantry_equipment
				amount = 500
			}
		}
		if = {
			limit = {
				has_equipment = {
					artillery_equipment > 250
				}
			}
			send_equipment = {
				target = FIN
				type = artillery_equipment
				amount = 250
			}
		}
		FIN = {
			country_event = { id = finland.5 days = 3 }
		}
	}

	option = {
		name = finland.3.c

		ai_chance = {

			factor = 1
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
			modifier = {
				factor = 100
				has_war = no
				has_democratic_government = yes
				strength_ratio = {
					tag = SOV
					ratio > 0.2
				}
				FIN = {
					has_democratic_government = yes
					strength_ratio = {
						tag = SOV 
						ratio > 0.3
					}
				}
			}			
		}

		declare_war_on = {
			target = SOV
			type = take_state
			generator = { 195 208 } # Leningrand and Luga
		}
		FIN = {
			country_event = { id = finland.6 days = 3 }
			add_opinion_modifier = {
				target = ROOT
				modifier = took_stand_for_us
			}
		}
	}
}

#Surrender when Soviet has taken too much land
country_event = {
	id = finland.7
	title = finland.7.t
	desc = finland.7.d
	picture = GFX_report_event_soviet_tanks_snow
	
	trigger = {
		owns_state = 146
		146 = { is_claimed_by = SOV }
		has_war_with = SOV
		surrender_progress > 0.1
		has_civil_war = no
		is_in_faction = no
	}

	mean_time_to_happen = { days = 10 }

	fire_only_once = yes

	#Surrender and offer territory
	option = {
		name = finland.7.a
		ai_chance = {
			base = 9
		}
		SOV = { country_event = { id = finland.8 } }
		hidden_effect = { set_country_flag = FIN_surrender }
	}

	#Never give up!
	option = {
		name = finland.7.b
		ai_chance = {
			base = 1
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
	}
}

#Soviet response to Finish surrender
country_event = {
	id = finland.8
	title = finland.8.t
	desc = finland.8.d
	picture = GFX_report_event_finnish_soldiers_snow
	
	is_triggered_only = yes

	#Accept surrender
	option = {
		name = finland.8.a
		ai_chance = {
			base = 9
		}
		if = {
			limit = {
				FROM = { original_tag = FIN }
			}
			white_peace = FROM
			diplomatic_relation = { country = FROM relation = non_aggression_pact }
			transfer_state = 146
			transfer_state = 147
			transfer_state = 1067
			hidden_effect = {
				add_state_core = 146
				add_state_core = 147
				remove_state_claim = 722
				remove_state_claim = 146
				remove_state_claim = 147
			}
		}
		if = {
			limit = {
				NOT = { FROM = { original_tag = FIN } }
			}
			white_peace = FROM
			diplomatic_relation = { country = FROM relation = non_aggression_pact }
			every_state = {
				limit = {
					is_core_of = FIN
					is_owned_by = FROM
				}
				transfer_state = PREV
			}
			hidden_effect = {
				remove_state_claim = 722
				remove_state_claim = 146
				remove_state_claim = 147
				add_state_core = 146
				add_state_core = 147
				add_state_core = 722
			}
		}
	}

	#Take it all
	option = {
		name = finland.8.b
		ai_chance = {
			base = 1
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}			
		}
		FROM = { country_event = { id = finland.9 } }
	}
}
