SOV_can_help_against_ENG = {
	SOV = {
		exists = yes
		has_government = communism
		has_civil_war = no
		NOT = { has_war_together_with = ENG }
	}
}

ROOT_will_help_RAJ = {
	ENG = {
		OR = {
			is_justifying_wargoal_against = ROOT
			has_war_with = ROOT
			any_allied_country = {
				OR = {
					is_justifying_wargoal_against = ROOT
					has_war_with = ROOT
				}
			}
			any_enemy_country = {
				any_enemy_country = {
					NOT = { has_war_with = ENG }
				}
				OR = {
					is_justifying_wargoal_against = ROOT
					has_war_with = ROOT
				}
			}
		}
	}
}
