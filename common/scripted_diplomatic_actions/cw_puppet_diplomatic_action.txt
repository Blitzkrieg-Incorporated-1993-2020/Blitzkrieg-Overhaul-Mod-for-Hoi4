scripted_diplomatic_actions = {
	lower_state_reliance = {
		icon = 25

		allowed = {
			always = yes
		}

		visible = {		#The target country
			is_subject_of = ROOT
		}

		selectable = {
			ROOT = { NOT = { has_country_flag = did_diplomatic_action } }		#Used to suspend request spam by AI			
		}

		can_be_accepted = {
			always = yes
		}

		can_be_sent = {
			always = yes
		}

		send_description = lower_state_reliance_LOC
		accept_title = lower_state_reliance_INITIATED_TITLE
		accept_description = lower_state_reliance_INITIATED_DESC
		
		cost = 25

		requires_acceptance = no

		on_sent_effect = {
		}

		complete_effect = {	
			if = {
				limit = {
					ROOT = { is_ai = yes }
				}
				ROOT = { set_country_flag = { flag = did_diplomatic_action value = 1 days = 45 } }
			}
			FROM = {
				add_autonomy_ratio = {
				    value = 0.05
				    localization = state_reliance_decreased
				}
			}	
		}

		ai_desire = {		
		}
	}
	raise_state_reliance = {
		icon = 25

		allowed = {
			always = yes
		}

		visible = {		#The target country
			is_subject_of = ROOT
		}

		selectable = {
			ROOT = { NOT = { has_country_flag = did_diplomatic_action } }		#Used to suspend request spam by AI
			FROM = {
				NOT = {
					OR = {
						has_autonomy_state = autonomy_imperial_dominion
						has_autonomy_state = autonomy_associate_republic
						has_autonomy_state = autonomy_free_regime
						has_autonomy_state = autonomy_collaborator_state
					}
				}
			}
		}

		can_be_accepted = {
			always = yes
		}

		can_be_sent = {
			always = yes
		}

		send_description = raise_state_reliance_LOC
		accept_title = raise_state_reliance_INITIATED_TITLE
		accept_description = raise_state_reliance_INITIATED_DESC
		
		cost = 50

		requires_acceptance = no

		on_sent_effect = {		
		}

		complete_effect = {
			if = {
				limit = {
					ROOT = { is_ai = yes }
				}
				ROOT = { set_country_flag = { flag = did_diplomatic_action value = 1 days = 45 } }
			}
			FROM = {
				add_autonomy_ratio = {
				    value = -0.1
				    localization = state_reliance_increased
				}
			}
		}

		ai_desire = {		
		}
	}
}
