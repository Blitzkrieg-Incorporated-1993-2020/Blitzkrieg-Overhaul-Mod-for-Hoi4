scripted_diplomatic_actions = {
	
	impose_trade_embargo = {

		allowed = {
			always = yes
		}
		visible = {
			ROOT = {
				NOT = {
					has_relation_modifier = {
						target = PREV
						modifier = embargo_static_modifier
					}
				has_idea = closed_economy
				is_subject_of = PREV
				}
			}
			PREV = {
				NOT = { is_subject_of = ROOT }
			}
		}

		selectable = {
			IF = {
				limit = {
					PREV = {
						OR = {
						has_war_together_with = ROOT 
						is_in_faction_with = ROOT 
						has_non_aggression_pact_with = ROOT 
						has_war_with = ROOT
						#is_subject_of = ROOT 
						}
					}
				}
				custom_trigger_tooltip = {
					tooltip = EMBARGO_DENIED
					always = no
				}
			}
			else = {
				custom_trigger_tooltip = {
					tooltip = EMBARGO_ALLOWED
					always = yes
				}
			}
		}

		cost = 10
		command_power = 0

		show_acceptance_on_action_button = no

		requires_acceptance = no

		accept_title = IMPOSE_EMBARGO_TARGET_TITLE
		accept_description = IMPOSE_EMBARGO_TARGET_DESC

		can_be_sent = {
			always = yes
		}

		icon = 1

		complete_effect = {
			ROOT = {
				add_opinion_modifier = { target = PREV modifier = new_emb }
				add_relation_modifier = { target = PREV modifier = embargo_static_modifier }
			}
			PREV = {
				add_opinion_modifier = { target = ROOT modifier = new_emb_opinion }
				remove_opinion_modifier = { target = ROOT modifier = emb_lifted }
			}
		}

		ai_desire = {
			base = 1
			modifier = { #game rule
				has_game_rule = {
					rule = allow_ai_embargo
					option = DONT_ALLOW_AI_EMBARGO
				}
			add = -5000
			}
			modifier = { #If opinion is lower than -50 they will impose embargo if no puppet 
				ROOT = {
					AND = {
						is_puppet = no
						has_opinion = {
							target = PREV
							value < -51
						}
					}
				}
			add = 50
			}

			modifier = { #If opinion is higher than -50 they won't impose embargo if no puppet
				ROOT = {
					AND = {
						is_puppet = no
						has_opinion = {
							target = PREV
							value > -49
						}
					}
				}
			add = -50
			}
			modifier = { #Counter Embargo
				PREV = {
					has_relation_modifier = {
						target = ROOT
						modifier = embargo_static_modifier
					}
				}
			add = 100
			}
			modifier = { #if opinion is higher than 50 they'll never impose embargo
				ROOT = {
					has_opinion = {
						target = PREV
						value > 49
					}
				}
			add = -100
			}
			modifier = { #if ally is at war with target they'll impose embargo regardless of autonomy state 
				ROOT = {
					any_allied_country = {
						has_war_with = PREV.PREV
					}
				}
				add = 300
			}

			modifier = { #if puppet they'll follow master #fixed DLC use now without any DLC
						AND = {
							ROOT = {
								#AND = {
								#	NOT = { has_autonomy_state = autonomy_dominion }
								#	NOT = { has_autonomy_state = autonomy_supervised_state }
								#	NOT = { has_autonomy_state = autonomy_satellite }
								#	NOT = { has_autonomy_state = autonomy_wtt_imperial_subject }
								#	is_subject = yes
								#}
								is_puppet = yes
							}
							any_other_country = {
								has_subject = ROOT
								has_relation_modifier = {
									target = PREV
									modifier = embargo_static_modifier
								}
							}
						}
			add = 300
			}
		}
	}

	cancel_trade_embargo = {

		allowed = {
			always = yes
		}

		visible = {
			ROOT = {
				has_relation_modifier = {
				target = PREV
				modifier = embargo_static_modifier
				}
			}
		}

		selectable = {
			IF = {
				limit = {
					ROOT = {
						has_relation_modifier = { target = PREV modifier = embargo_static_modifier }
					}
					PREV = {
						AND = {
							OR = {
							is_subject_of = ROOT
							NOT = { has_war_with = ROOT }
							}
						}
					}
				}
				custom_trigger_tooltip = {
					tooltip = CANCEL_EMBARGO_ALLOWED
					always = yes
				}
			}
			else = {
				custom_trigger_tooltip = {
					tooltip = CANCEL_EMBARGO_DENIED
					always = no
				}
			}
		}

		cost = 10
		command_power = 0

		requires_acceptance = no

		show_acceptance_on_action_button = no

		accept_title = CANCEL_EMBARGO_TARGET_TITLE
		accept_description = CANCEL_EMBARGO_TARGET_DESC

		can_be_sent = {
			always = yes
		}

		icon = 1

		complete_effect = {
			ROOT = {
				remove_opinion_modifier = { target = PREV modifier = new_emb }
				
				remove_relation_modifier = { target = PREV modifier = embargo_static_modifier }
			}
			PREV = {
			add_opinion_modifier = { target = ROOT modifier = emb_lifted }
			remove_opinion_modifier = { target = ROOT modifier = new_emb_opinion }
			}
		}

		ai_desire = {
			base = 1

			modifier = { #game rule
				has_game_rule = {
					rule = allow_ai_embargo
					option = DONT_ALLOW_AI_EMBARGO
				}
				add = -5000
			}
			modifier = { #if opinion is higher -50 they'll cancel emb if not puppet
				ROOT = {
					AND = {
						is_puppet = no
						has_opinion = {
							target = PREV
							value > -51
						}
					}
				}
				add = 50
			}
			modifier = { #molotov_ribbentrop
        		has_global_flag = sov_yes_pact
				ROOT = {
					OR = {
						AND = {
							tag = GER
							PREV = { tag = SOV }
						}
						AND = {
							tag = SOV
							PREV = { tag = GER }
						}
					}
				}
				add = 500
			}
			modifier = { #if opinion is lower -50 they won't cancel if not puppet
				ROOT = {
					AND = {
						is_puppet = no
						has_opinion = {
							target = PREV
							value < -49
						}
					}
				}
			add = -50
			}
			modifier = { #if ally has war with target they won't cancel
				ROOT = {
					any_allied_country = {
						has_war_with = PREV.PREV
					}
				}
			add = -300
			}

			modifier = { #see above #won't cancel if master has embargo
						AND = {
							ROOT = {
								#AND = {
								#	NOT = { has_autonomy_state = autonomy_dominion }
								#	NOT = { has_autonomy_state = autonomy_supervised_state }
								#	NOT = { has_autonomy_state = autonomy_satellite }
								#	NOT = { has_autonomy_state = autonomy_wtt_imperial_subject }
								#	is_subject = yes
								#}
								is_puppet = yes
							}
							any_other_country = {
								has_subject = ROOT
								has_relation_modifier = {
									target = PREV
									modifier = embargo_static_modifier
								}
							}
						}
			add = -300
			}

			modifier = { #this is used to lift embargo when master has lifted
				AND = {
					ROOT = {
						AND = {
							is_puppet = yes
							has_relation_modifier = {
								target = PREV
								modifier = embargo_static_modifier
							}
						}
					}
					any_other_country = {
						has_subject = ROOT
						NOT = {
							has_relation_modifier = {
								target = PREV
								modifier = embargo_static_modifier
							}
						}
					}
				}
			add = 300
			}

			modifier = { #Counter Embargo
				PREV = {
					has_relation_modifier = {
						target = ROOT
						modifier = embargo_static_modifier
					}
				}
			add = -300
			}
		}
	}
}
